# JoystickShield 按键映射配置文件
# 格式: Arduino消息 = 键盘按键

# 摇杆方向映射 (WASD 游戏控制)
摇杆：上 = w
摇杆：下 = s
摇杆：左 = a
摇杆：右 = d
摇杆：左上 = w,a
摇杆：右上 = w,d
摇杆：左下 = s,a
摇杆：右下 = s,d

# 按钮映射
摇杆按键按下 = space
上按钮按下 = up
下按钮按下 = down
左按钮按下 = left
右按钮按下 = right
E 按钮按下 = e
F 按钮按下 = f

# 串口设置
[SERIAL]
baudrate = 115200
timeout = 1

# 高级设置
[SETTINGS]
# 按键释放延迟 (毫秒)
key_release_delay = 50
# 摇杆死区 (忽略小幅度移动)
deadzone_threshold = 10
